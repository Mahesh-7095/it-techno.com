HTML Web Storage

Using localStorage

localStorage allows you to store data persistently in the browser. 


<input type="text" id="localStorageInput" placeholder="Enter text">
<button onclick="saveToLocalStorage()">Save to localStorage</button>
<button onclick="loadFromLocalStorage()">Load from localStorage</button>
<p id="localStorageOutput"></p>
<script>
    function saveToLocalStorage() {
        const input = document.getElementById("localStorageInput").value;
        localStorage.setItem("savedText", input);
    }
    function loadFromLocalStorage() {
        const output = localStorage.getItem("savedText");
        document.getElementById("localStorageOutput").innerText = output || "No data found.";
    }
</script>
